export function Card(props) {
  const emailprefix="mailto:"
  const pagelink=props.page
  const imglink=props.pfplink
  const name=props.name
  const description=props.description
  const address=props.address
  const email=props.email
  if (pagelink !== undefined ) {
    return <>
        <div className="flex mb-4 w-full">
          <div className='p-4 w-1/2'>
            <Link href={{pathname: pagelink}}>
              <Image alt={name} src={imglink} width={1000} height={1000}/>
            </Link>
          </div>
          <div className="p-4 w-1/2">
            <p><b>{name}</b></p>
            <Link href={emailprefix.concat(email)}><p className="text-red-700 hover:text-red-400">Email</p></Link>
            <Link href={{pathname: pagelink}}><p className="text-red-700 hover:text-red-400">Profile Page↗️</p></Link>
            <p>{description}</p>
            <i className='text-sm'>{address}</i>
          </div>
        </div>
        </>
  }
  else {
    return <>
        <div className="flex mb-4 w-full">
          <div className='p-4 w-1/2'>
            <Link href={{pathname: pagelink}}>
              <Image alt={name} src={imglink} width={1000} height={1000}/>
            </Link>
          </div>
          <div className="p-4 w-1/2">
            <p><b>{name}</b></p>
            <Link href={emailprefix.concat(email)}><p className="text-red-700 hover:text-red-400">Email</p></Link>
            <p>{description}</p>
            <i className='text-sm'>{address}</i>
          </div>
        </div>
        </>
  }
  
}
